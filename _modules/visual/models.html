[docs]class Placement(models.Model):
    """
    Stores ball placement coordinates with foreign key representing
    one-sysnet-to-many-placement relationship.
    """
    sysnet = models.ForeignKey(SysNetlist, on_delete=models.CASCADE)
    cx = models.IntegerField()
    cy = models.IntegerField()
    bucket = models.CharField(max_length=300, blank=True)

    def __unicode__(self):
        return str(self.sysnet) + " " + str(self.cx) + "," + str(self.cy)

    def __str__(self):
        return unicode(self).encode('utf-8')
