[docs]def all_chiplets(request):
    """
    Shows a list of all chiplets in the database
    """
    chiplet_list = []
    chiplets = list(Chiplet.objects.all())
    for chip in chiplets:
        chiplet_obj = {}
        chiplet_obj["mpn"] = chip.mpn
        chiplet_obj["cid"] = chip.cid
        chiplet_obj["x_dim"] = chip.x_dim
        chiplet_obj["y_dim"] = chip.y_dim
        chiplet_obj["z_dim"] = chip.z_dim
        chiplet_obj["num_balls"] = chip.num_balls
        chiplet_obj["io_count"] = chip.get_io_count()
        chiplet_obj["cfg_count"] = chip.get_cfg_count()
        chiplet_list.append(chiplet_obj)

    return render(request, 'chiplet/chipletList.html', {
        'chipletList': chiplet_list
    })

[docs]def add_chiplet(request):
    """
    Function to add a chiplet - renders a chiplet form if 'GET' request
    is received, and saves a chiplet if 'POST' request is received.
    """
    if request.method == 'POST':
        form = ChipletForm(request.POST)
        if form.is_valid():
            chiplet = form.save(commit=False)
            post_dict = request.POST.items()
            fields = parse_request_io(post_dict)
            save_chiplet_ios(chiplet, fields)
            return all_chiplets(request)
        else:
            print form.errors
    else:
        form = ChipletForm()
    return render(request, 'chiplet/addChiplet.html', {'form': form})



[docs]def add_cfg(request, cid):
    """
    Function to add a config - renders a list of configs if 'GET' request
    is received, and saves a config if 'POST' request is received.
    """
    if request.method == "POST":
        post_dict = request.POST.items()
        configname = request.POST["configname"]
        proto = request.POST["proto"]
        cfg_dict = parse_request_cfg(post_dict)
        save_cfg(cid, cfg_dict, configname, proto)
        return show_cfgs(request, cid)
    else:
        c = Chiplet.objects.get(cid=cid)
        io_list = c.get_ios()
    return render(request, 'chiplet/addCfg.html', {'chiplet': c, 'ioList': io_list})

[docs]def parse_request_io(post_dict):
    """
    Parse POST request dictionary of IOs to get IO object attributes.
    Args:
        post_dict: post dictionary
    Returns:
        dictionary of IOs belonging to a chiplet
        e.g.
        {
            "0": {"pnum": "A1",
                  "pname": "VDD",
                  ...},
            "1": {"pnum" : "A2",
                  "pname": "VSS",
                  ...},
            ...
        }
    To Do:
        refactor to generalize parse_request functions for reuse
    """
    fields = {}
    for k, v in post_dict:
        try:
            # try if the post input is for the IO (starts with a number)
            int(k[0])
            # split string number#io_attr
            klist = k.split('+')
            num = klist[0]
            attr = klist[1]
            if num in fields:
                fields[num][attr] = v
            else:
                obj = {}
                obj[attr] = v
                fields[num] = obj
        except ValueError:
            pass

    return fields

[docs]def save_chiplet_ios(chiplet, io_dict):
    """
    Save chiplet and corresponding IOs into database
    Args:
        chiplet: chiplet object from form
        io_dict: dictionary of IOs and their attributes
    """
    chiplet.save()
    for num in io_dict:
        io = IO()
        k = io_dict[num]
        io.chiplet = chiplet
        io.pname = k["pname"]
        io.pnum = k["pnum"]
        io.sigt = k["sigt"]
        io.io_mech = k["io_mech"]
        io.diameter_mm = k["diameter_mm"]
        io.x_pos_mm = k["x_pos_mm"]
        io.y_pos_mm = k["y_pos_mm"]
        io.save()



[docs]def save_cfg(cid, cfg_dict, configname, proto):
    """
    Save config and corresponding ConfigIOs into database
    Args:
        cid: chiplet ID
        cfg_dict: dictionary of ConfigIOs and their attributes
        configname: name of config specified by user in form
        proto: protocol of config specified by user in form
    """
    chiplet = Chiplet.objects.get(cid=cid)
    c = Config()
    c.cfgname = configname
    c.proto = proto
    c.chiplet = chiplet
    c.save()
    for i in cfg_dict:
        v = cfg_dict[i]
        cfgio = ConfigIO()
        cfgio.io = IO.objects.get(chiplet=chiplet, pnum=i)
        cfgio.cfg = c
        cfgio.tpu = v["tpu"]
        cfgio.tpd = v["tpd"]
        cfgio.tps = v["tps"]
        cfgio.sysnn = v["sysnn"]
        cfgio.driver = v["driver"]
        cfgio.fastnn = v["fastnn"]
        cfgio.fastconn = v["fastconn"]
        cfgio.rdlconn = v["rdlconn"]
        cfgio.ioconn = v["ioconn"]
        cfgio.rpriority = v["rpriority"]
        cfgio.rdl2tile = v["rdl2tile"]
        cfgio.pass_0201 = v["pass_0201"]
        cfgio.pass_0402 = v["pass_0402"]
        cfgio.save()
