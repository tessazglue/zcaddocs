[docs]    def place_system(self):
        """
        Create Connector object to get respective data from db,
        rotate and shift each chiplet to its center on grid,
        generate grid & ubump mapping
        Returns:
            final_coords: coordinates of all filled ubumps in the system
        """
        db_conn = Connector(self.sys_id)
        sids = db_conn.get_syscomp_ids()
        cens = db_conn.get_system_cens()
        rots = db_conn.get_system_rots()
        dims = db_conn.get_system_dims()
        num_chips = len(sids)

        for idx in range(0,num_chips):
            s_coords = db_conn.get_coords_df(sids[idx])
            coordlist = self._rotate_and_shift_center(s_coords, rots[idx], cens[idx], dims[idx])
            self.coords = self.coords.append(coordlist)

        self.coords = self.coords.reset_index(drop=True)
        grid_coords = self._generate_grid(self.coords)
        final_coords = self._df_to_dict(grid_coords)
        return final_coords




[docs]    def _rotate_and_shift_center(self, s_coords, rot, cen, dim):
        """
        Pre-process dataframe by splitting into reference details and
        xy coordinates, rotate chiplet, snap to grid, and shift chiplet to
        its center on the grid.
        Args:
            s_coords: dataframe with cid, io_ref, ball_dia, cen_x, cen_y columns
            rot: rotation integer in multiples of 90, of a chiplet
            cen: list containing x and y coordinates of center of a chiplet
            dim: list containing length and width of a chiplet
        Returns:
            coordlist: dataframe of ref details, new center coords and dia
        """
        new_coord = pd.DataFrame()
        # split into io ref details & coordinates info
        ref_details = s_coords[s_coords.columns[0:2]]
        ball_dia = s_coords[s_coords.columns[2]]
        c_xy = s_coords[s_coords.columns[3:5]]
        # rotation
        ang = math.radians(rot)
        r = [[math.cos(ang), -math.sin(ang)], [math.sin(ang), math.cos(ang)]]
        c_xy = c_xy.dot(r)
        # snap to grid
        difference_to_move = self._snap_to_grid(c_xy)
        # shift to center
        new_coord["cen_x"] = c_xy[c_xy.columns[0]] + cen[0] - difference_to_move[0]
        new_coord["cen_y"] = c_xy[c_xy.columns[1]] + cen[1] - difference_to_move[1]
        coordlist = pd.concat([ref_details, new_coord, ball_dia], axis=1)
        return coordlist




[docs]    def _snap_to_grid(self, c_xy):
        """
        Snap to grid means moving the chiplet balls to a position where its
        CENTER either lies on the center of a tile, or touches either of the
        4 sides of a tile. First ball of a chiplet will be used to snap to grid,
        assuming that all the other balls of that chiplet will also land
        properly.
        Args:
            c_xy: dataframe of all ios' x y coordinates of a chiplet
        Returns:
            difference_to_move: a list of distance that a chiplet's center has
                                to move horizontally, and vertically.
        """
        first_ball = self._get_first_ball(c_xy)
        snapped_ball_X = self._round_coord(first_ball[0])
        snapped_ball_Y = self._round_coord(first_ball[1])
        return [first_ball[0] - snapped_ball_X, first_ball[1] - snapped_ball_Y]




[docs]    def _get_first_ball(self, c_xy):
        """
        The first ball is defined to be the top most and left most ball of
        a chiplet.
        Args:
            c_xy: dataframe of all ios' x y coordinates of a chiplet
        Returns:
            coordlist: list of x,y coordinates of first ball
        """
        highest_ball = c_xy[c_xy[1] == c_xy[1].max()]
        first_ball = highest_ball[highest_ball[0] == highest_ball[0].min()]
        first_ball = first_ball.reset_index(drop=True)
        return [first_ball[0][0], first_ball[1][0]] # get first ball in leftover list




[docs]    def _round_coord(self, coord):
        """
        Round a coordinate to the nearest half tile, and returns the newly
        snapped coordinate.
        Args:
            coord: a coordinate for rounding (may be negative)
        Returns:
            new_coord: newly snapped coordinate
        Example for negative number -92:
            -92 % 50 => -(92 % 50) => -42 => -92 - 50-(-42)
        """
        trail_num = coord % 50
        if trail_num == 0:
            return coord
        elif trail_num >= 25: # round up
            return coord + 50 - trail_num
        else: # round down
            return coord - trail_num




[docs]    def _generate_grid(self, coords):
        """
        Generate grid for returning, and for creating csv files.
        Args:
            coords: dataframe of ref details, new center coords and dia, of
                    all chiplets in the system
        Returns:
            grid_coords: dataframe of coordinates of all filled ubumps
                         in the system
        """
        #coords.to_csv('data/test/pre_generate_grid.csv', index=False)
        bound_ios = self._generate_upper_lower_bound(coords)
        valid_bound_ios = self._remove_balls_out_of_range(bound_ios)
        is_gridmap_output = self._output_gridmap_csv(valid_bound_ios)
        grid_coords = self._get_grid_coords(valid_bound_ios)
        #grid_coords.to_csv('data/test/post_generate_grid.csv', index=False)
        return grid_coords




[docs]    def _generate_upper_lower_bound(self, coords):
        """
        Generate upper and lower bounds of each IO in system. As long as an
        IO fully or partially covers a tile, a ubump will be assigned to that
        IO.
        Args:
            coords: dataframe of ref details, new center coords and dia, of
                    all chiplets in the system (cid,io_ref,cen_x,cen_y,ball_dia)
        Returns:
            bound_ios: dataframe of (cid,io_ref,X_LB,X_UB,Y_LB,Y_UB)
        """
        #coords.to_csv("data/test/coords_df.csv", index=False)
        ball_x_y = coords[coords.columns[2:5]]
        ref_details = coords[coords.columns[0:2]]
        bound_ios = pd.DataFrame()
        # create upper and lower bounds of each IO
        b_xlb = [0.01, 0, -0.005]
        b_xub = [0.01, 0, 0.005]
        b_ylb = [0, 0.01, -0.005]
        b_yub = [0, 0.01, 0.005]
        bound_ios["X_LB"] = ball_x_y.dot(b_xlb)
        bound_ios["X_UB"] = ball_x_y.dot(b_xub)
        bound_ios["Y_LB"] = ball_x_y.dot(b_ylb)
        bound_ios["Y_UB"] = ball_x_y.dot(b_yub)
        bound_ios["X_LB"] = (bound_ios["X_LB"] + 0.5).apply(lambda x: np.ceil(round(x,1)))
        bound_ios["X_UB"] = (bound_ios["X_UB"] + 0.5).apply(lambda x: np.floor(round(x,1)))
        bound_ios["Y_LB"] = (bound_ios["Y_LB"] + 0.5).apply(lambda x: np.ceil(round(x,1)))
        bound_ios["Y_UB"] = (bound_ios["Y_UB"] + 0.5).apply(lambda x: np.floor(round(x,1)))
        # merge both ref and processed bound ios
        merged_bound_ios = pd.concat([ref_details, bound_ios], axis=1)
        #merged_bound_ios.to_csv("data/test/bound_ios.csv", index=False)
        return merged_bound_ios




[docs]    def _remove_balls_out_of_range(self, coords):
        """
        Remove balls that are out of range of the grid (0-36,0-56).
        Args:
            coords: dataframe of X_LB, X_UB, Y_LB, Y_UB & ref details
        Returns:
            coords: dataframe of valid X_LB, X_UB, Y_LB, Y_UB & ref details
        """
        #coords.to_csv('data/test/pre_bound_ios_on_right_of_range.csv', index=False)
        coords = coords[coords['X_LB'] > 0]
        coords = coords[coords['X_UB'] > 0]
        coords = coords[coords['Y_LB'] > 0]
        coords = coords[coords['Y_UB'] > 0]
        coords = coords[coords['X_LB'] < 36]
        coords = coords[coords['X_UB'] < 36]
        coords = coords[coords['Y_LB'] < 56]
        coords = coords[coords['Y_UB'] < 56]
        coords = coords.reset_index(drop=True)
        #coords.to_csv('data/test/post_bound_ios_on_right_of_range.csv', index=False)
        return coords




[docs]    def _output_gridmap_csv(self, coords):
        """
        Generate gridmap csv, and ubumpid mapping csv.
        Args:
            coords: dataframe of X_LB, X_UB, Y_LB, Y_UB & ref details
        Returns:
            is_valid: boolean, whether csv generation was successful
        """
        ubump_mapping = coords[coords.columns[0:2]]
        ubump_mapping['ubump'] = range(1, len(ubump_mapping) + 1)
        ubump_mapping.to_csv('./data/placement_out/ubump_' + str(self.sys_id) + '.csv', index=False)

        GRIDMAP_X = 56
        GRIDMAP_Y = 36
        gridmap = np.zeros((GRIDMAP_X, GRIDMAP_Y))
        num_balls = len(coords.index)

        for ball in range(0, num_balls):
            for xcoord in range(int(coords["X_LB"][ball])-1, int(coords["X_UB"][ball])):
                for ycoord in range(int(coords["Y_LB"][ball])-1, int(coords["Y_UB"][ball])):
                    try:
                        gridmap[ycoord][xcoord] = ubump_mapping["ubump"][ball]
                    except:
                        pass

        gridmap = pd.DataFrame(gridmap)
        # start row and column index from 1
        gridmap.index = gridmap.index + 1 # row
        gridmap.columns = gridmap.columns.to_series().apply(lambda x: x+1) # column
        # flip grid on x axis for viewing
        gridmap = gridmap.iloc[::-1]
        gridmap.to_csv('./data/placement_out/placement_' + str(self.sys_id) + '.csv')

        return True




[docs]    def _get_grid_coords(self, coords):
        """
        Generate grid coordinates for returning. Assume that cid and pnum are
        strings without spaces.
        Args:
            coords: dataframe of X_LB, X_UB, Y_LB, Y_UB & ref details
        Returns:
            final_coords: dataframe of ball index, x and y coordinates
        """
        #coords.to_csv('data/test/pre_get_grid_coords.csv', index=False)
        num_balls = len(coords.index)
        final_coords = pd.DataFrame()
        for ball in range(0, num_balls):
            col_range = range(int(coords["X_LB"][ball])-1, int(coords["X_UB"][ball]))
            row_range = range(int(coords["Y_LB"][ball]), int(coords["Y_UB"][ball])+1)
            ball_coords = pd.DataFrame(list(product(row_range, col_range)), columns=['x', 'y'])
            ball_coords["ball"] = coords["cid"][ball] + " " + coords["io_ref"][ball]
            ball_coords["ubumpid"] = ball + 1 # +1 to start from index 1
            ball_coords = ball_coords[["ball", "x", "y", "ubumpid"]]
            final_coords = final_coords.append(ball_coords)

        final_coords = final_coords.reset_index(drop=True)
        #final_coords.to_csv('data/test/post_get_grid_coords.csv', index=False)
        return final_coords




[docs]    def _df_to_dict(self, coords):
        """
        Convert dataframe to dictionary of final coordinates
        Args:
            coords: dataframe consisting of all coordinates and ball diameter
        Returns:
            coord_dict: dictionary of each ball's coordinates
            example: {"U2 A1": [{"cx": 123, "cy": 356, "ubump": 1},
                                {"cx": 124, "cy": 356, "ubump": 1}],
                      "U2 A2": [{"cx": 128, "cy": 358, "ubump": 2},...]}
        """
        coord_dict = {}
        for index, row in coords.iterrows():
            if row[0] in coord_dict:
                coord_dict[row[0]].append({'cx':row[1], 'cy':row[2], 'ubump': row[3]})
            else:
                coord_dict[row[0]] = [{'cx':row[1], 'cy':row[2], 'ubump': row[3]}]
        return coord_dict
