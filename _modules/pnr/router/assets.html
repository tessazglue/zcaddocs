    """
    Representation of the whole tilegrid and its individual tile associations

    Attributes:
        rows - list[Row()] list of Row objects on the Tile Grid
        cols - list[Column()] list of Column objects on the Tile Grid
        tiles - list[list[Tile()]] list of Tile objects on the Tile Grid
        num_cols_filled - [INT] number of columns used

    """

    def __init__(self):
        self.rows = [self.Row() for _ in xrange(Constant.ROW_SIZE)]
        self.cols = [self.Column() for _ in xrange(Constant.COL_SIZE)]
        self.tiles = [[self.Tile() for _ in xrange(Constant.COL_SIZE)]
                      for _ in xrange(Constant.ROW_SIZE)]
        self.num_cols_filled = 0
        self.is_filled = False

[docs]    def occupy_rowbit(self, row_index, row_bit, col_index, fnn):
        """Associate the rowbit to given col_index and fnn
        Args:
            row_index (int)
            row_bit (int)
            col_index (int)
            fnn (int)
        Returns:
        """

        rowbit = self.rows[row_index].bits[row_bit]
        if not rowbit.is_occupied:
            self.rows[row_index].num_bits_filled += 1
            self.rows[row_index].check_bits()

            rowbit.is_occupied = True
            rowbit.fnn = fnn
            rowbit.is_connected = True
            rowbit.col_index = col_index
        return



[docs]    def get_rowbit_association(self, row_index, col_index, fnn):
        """Obtains the rowbit associated to given row_index, col_index and fnn
        Args:
            row_index (int)
            col_index (int)
            fnn (int)
        Returns:
            int: the rowbit associated to fnn. -1 if not found
        """

        ind0 = col_index % Constant.ROWBIT_MODULO
        ind1 = col_index % Constant.ROWBIT_MODULO + Constant.ROWBIT_MODULO
        bit0 = self.rows[row_index].bits[ind0]
        bit1 = self.rows[row_index].bits[ind1]

        if bit0.is_occupied and bit0.fnn == fnn and \
                bit0.is_connected and bit0.col_index == col_index:
            return ind0
        if bit1.is_occupied and bit1.fnn == fnn and \
                bit1.is_connected and bit1.col_index == col_index:
            return ind1
        return -1

        """Associates the col_index to given fnn

        """

[docs]    def get_col_association(self, fnn):
        """Obtains the col_index associated to the given fnn
        Args:
            fnn (int)
        Returns:
            int: the col_index associated to the fnn. -1 if not found.
        """

        for col in self.cols:
            if col.is_occupied and col.fnn == fnn:
                return self.cols.index(col)
        return -1

[docs]    def grid_to_string(self):
        """Get information about occupied rows and columns
        Args:
        Returns:
             str
        """

        result = ["\nRows---------------\n"]
        for row in xrange(Constant.ROW_SIZE):
            for bit in xrange(Constant.ROWBIT_SIZE):
                rowbit = self.rows[row].bits[bit]
                if rowbit.is_occupied:
                    result.append("Row:%d Bit:%d\t%s\n" % (row, bit, str(rowbit)))
        result.append("\nColumns---------------\n")
        for col in xrange(Constant.COL_SIZE):
            column = self.cols[col]
            result.append("Col:%d\t%s\n" % (col, str(column)))
        return "".join(result)



[docs]    def tile_to_string(self):
        """Get information about non-default tiles
        Args:
        Returns:
            str
        """

        result = ["\nTiles NonZero Presets---------------\n"]
        for row in xrange(Constant.ROW_SIZE):
            for col in xrange(Constant.COL_SIZE):
                if not self.tiles[row][col].is_default():
                    result.append("R:{:<3} C:{:<3} {}".format(
                        row, col, self.tiles[row][col].to_string()))
        return "\n".join(result)


    def __str__(self):
        result = []
        for row in xrange(Constant.ROW_SIZE):
            for col in xrange(Constant.COL_SIZE):
                tile = self.tiles[row][col]
                if tile.is_occupied:
                    result.append("%d, %d, BID=%d, FNN=%d" % (row, col, tile.bid, tile.fnn))
        return "\n".join(result)

[docs]    class Tile(object):
        """
        Representation of a single tile and its ownership associations

        Attributes:
            is_occupied - [BOOLEAN] if tile is occupied by a bump
            bid - [INT] uBumpID associated with tile
            fnn - [INT] FastNetName associated with tile
            gridcode - [STRING] programming code associated with tile
        """

        def __init__(self, is_occupied=False, bid=0, fnn=0, preset="0000000000"):
            self.is_occupied = is_occupied
            self.bid = bid
            self.fnn = fnn
            self.gridcode = preset

        def __set_gridcode(self, value,
                           start=TileConstant.GRIDCODE_OFF_START,
                           end=TileConstant.GRIDCODE_OFF_END):
            self.gridcode = self.gridcode[:start] + value[::-1] + self.gridcode[end:]
            return

        def __get_gridcode(self,
                           start=TileConstant.GRIDCODE_OFF_START,
                           end=TileConstant.GRIDCODE_OFF_END):
            return self.gridcode[start:end][::-1]

[docs]        class RowBit(object):
            """
            Representation of a single RowBit and its ownership associations

            Attributes:
                is_occupied - [BOOLEAN] if bit is associated
                fnn - [INT] FastNetName using the bit connection
                is_connected - [BOOLEAN] if bit is connected
                col_index - [INT] column index using the rowbit connection
            """

            def __init__(self, is_occupied=False, fnn=0, is_connected=False, col_index=0):
                self.is_occupied = is_occupied
                self.fnn = fnn
                self.is_connected = is_connected
                self.col_index = col_index

            def __str__(self):
                return "{occupy} {connect}".format(
                    occupy="Occupied by {}".format(
                        self.fnn) if self.is_occupied else "Not Occupied",
                    connect="and Connected to col {}".format(
                        self.col_index) if self.is_connected else "and Not Connected to Col")
