        """Connect all tiles connected to balls with FNN:GND to GND
        Args:
        Returns:
        """

        bid_list = self.fnn_to_bid[Constant.FNN_GND]
        for bid in bid_list:
            # Print Bumps connected
            for coord in self.bid_to_coord[bid]:
                self.tilegrid.tiles[coord.row][coord.col].connect_gnd()

[docs]    def set_pull_value(self, res, bid, is_pup):
        """Determining logic of resistance level on FAST board
        Args:
            res (int):
            bid (int):
            is_pup (bool): whether it's pull-up or pull-down
        Returns:
        """

        coord_list = self.bid_to_coord[bid]

        if res == RouterConstant.RES_0:
            self.pull_conn(coord_list[0], 1, is_pup)
        elif RouterConstant.RES_1_4_ < res < RouterConstant.RES_1__4:
            if len(coord_list) < 4:
                raise RoutingError("Unable to set pull values")
            for index in xrange(4):
                self.pull_conn(coord_list[index], 2, is_pup)
        elif RouterConstant.RES_1_3_ <= res <= RouterConstant.RES_1__3:
            if len(coord_list) < 3:
                raise RoutingError("Unable to set pull values")
            for index in xrange(3):
                self.pull_conn(coord_list[index], 2, is_pup)
        elif res == RouterConstant.RES_1:
            self.pull_conn(coord_list[0], 2, is_pup)
        elif res == RouterConstant.RES_2:
            self.pull_conn(coord_list[0], 3, is_pup)
        else:
            raise RoutingError("Resistance Level not supported")
        return



[docs]    def pull_conn(self, coord, code, is_pup):
        """Set pull values accordingly on FAST
        Args:
            coord (BumpCoord)
            code (int)
            is_pup (bool): pull-up or pull-down
        Returns:
        """

        if is_pup:
            self.tilegrid.tiles[coord.row][coord.col].connect_pup(code)
        else:
            self.tilegrid.tiles[coord.row][coord.col].connect_pdn(code)
        return



[docs]    def route_bumps(self):
        """Wrapper method to make all routing calls
        Includes both analog and digital routing (ONLY ANALOG NOW)
        Args:
        Returns:
        """

        priority_tuples = self.get_route_priority()
        # 3 possible methods to call:
        # route_analog
        # route_digital
        # route_i2c

        for tup in priority_tuples:
            for index in range(1, len(tup)):
                fnn = tup[index]
                bid_list = self.fnn_to_bid[fnn]
                # ANALOG ROUTIING
                # This will be changed to fnn % 32 / 8 == 0
                if fnn % 100 == 0:
                    if len(bid_list) != 1:
                        raise RoutingError("%d should route to tile only" % fnn)
                else:
                    if len(bid_list) == 1:
                        raise RoutingError("%d should route to other bumps only" % fnn)
                self.route_analog(bid_list)



[docs]    def get_route_priority(self):
        """Preprocess: Sort all bumps by priority first before routing
        Args:
        Returns:
            list: FNN connection sets sorted by priority
        """

        priority_to_fnn = {}
        priority_tuples = []
        for fnn, bid_list in self.fnn_to_bid.items():
            # This will be changed to fnn & 32 == 1
            if fnn >= Constant.FNN_FAST_MIN:
                max_priority = 0
                for bid in bid_list:
                    ball = self.bid_to_ball[bid]
                    if ball.rprio > max_priority:
                        max_priority = ball.rprio

                if max_priority in priority_to_fnn:
                    priority_to_fnn[max_priority].append(fnn)
                else:
                    priority_to_fnn[max_priority] = [fnn]

        for priority, fnn_list in priority_to_fnn.items():
            tup = (priority,)
            for fnn in fnn_list:
                tup += (fnn,)
            priority_tuples.append(tup)

        priority_tuples.sort(key=lambda t: t[0], reverse=True)
        return priority_tuples



[docs]    def route_analog(self, bid_list):
        """Set configuration of tiles that are connected to the ball to route them.
        If there are multiple balls, a driver will be chosen (logic TBD)
        Choose the first BumpCoord's col occupy col on tilegrid - col_index
        For every BumpCoord, occupy rowbit (associated to col_index),
        Then connect row and col

        Errors:
            When there is no more column to occupy
            When there is no more rowbit to assign to fnn [Decision TBD]
        Args:
            bid_list (list): list of bid to be routed (either single or multi)
        Returns:
        """

        for bid in bid_list:
            select_ball = self.bid_to_ball[bid]
            # Unimplemented
            # if select_ball.is_driver:
            #    break
        coord_list = self.bid_to_coord[select_ball.bid]
        coord = coord_list[0]

        col_index = self.get_next_avail_col(coord.col, select_ball.fnn)
        if col_index < 0:
            raise RoutingError("TileGrid has filled all columns available")

        self.tilegrid.occupy_col(col_index, select_ball.fnn)
        for bid in bid_list:
            select_ball = self.bid_to_ball[bid]
            coord_list = self.bid_to_coord[bid]
            for coord in coord_list:
                row_index = coord.row
                bit_index = self.get_next_avail_rowbit(row_index, col_index, select_ball.fnn)
                if bit_index < 0:  # FIXME: Decision if we should delay routing
                    raise RoutingError("Failed to connect %s %s FNN %d" %
                                       (select_ball.mpn, select_ball.pnum, select_ball.fnn))

                self.tilegrid.occupy_rowbit(row_index, bit_index, col_index, select_ball.fnn)

                bit_code = 2 if bit_index >= Constant.ROWBIT_MODULO else 1
                col_code = self.tilegrid.tiles[row_index][col_index].get_col_value()
                col_code |= bit_code
                self.tilegrid.tiles[row_index][col_index].connect_col(col_code)
                self.tilegrid.tiles[coord.row][coord.col].connect_row(bit_index)
        return



[docs]    def get_next_avail_col(self, col_index, fnn):
        """Finds the next available column to associate to FNN.
        Increments to alternating side to search for a free column
        Args:
            col_index (int)
            fnn (int)
        Returns:
            int: column index that is available, or one that is already associated to fnn.
        """

        index = self.tilegrid.get_col_association(fnn)
        if index > -1:
            return index

        if self.tilegrid.is_filled:
            return -1

        iter_ = 0
        while True:
            index = col_index + alternate_increment(iter_)
            if index < 0 or index >= Constant.COL_SIZE:
                continue

            if not self.tilegrid.cols[index].is_occupied:
                return index

            iter_ += 1



[docs]    def get_next_avail_rowbit(self, row_index, col_index, fnn):
        """Finds the next available rowbit to associate to FNN
        Rowbit to be assigned is determined by col_index
            - rowbit = col_index % (.5 available bits) + (.5 available bits)
        Args:
            row_index (int)
            col_index (int)
            fnn (int)
        Returns:
            int: the rowbit that is available, or one that is already associated,
                or -1 if not found
        """

        index = self.tilegrid.get_rowbit_association(row_index, col_index, fnn)
        if index > -1:
            return index

        if self.tilegrid.rows[row_index].is_filled:
            return -1

        ind0 = col_index % Constant.ROWBIT_MODULO
        ind1 = col_index % Constant.ROWBIT_MODULO + Constant.ROWBIT_MODULO
        if not self.tilegrid.rows[row_index].bits[ind0].is_occupied:
            return ind0
        if not self.tilegrid.rows[row_index].bits[ind1].is_occupied:
            return ind1

        return -1




[docs]def alternate_increment(iter_):
    """Simple alternating increment function
    Args:
        iter_ (int)
    Returns:
        int: -ve if iter_ is even, +ve if iter_ is odd
    """

    if iter_ % 2 == 0:
        return -(iter_ >> 1)
    return (iter_ >> 1) + 1
