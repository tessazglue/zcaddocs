    """
    Writes routed TileGrid information to visualisation and tile programming files

    FAST Visualisation:
    $_row.gph
    $_col.gph
    $_uBump.gph

    FAST Preset Verilog and Data:
    $_presets.vs
    $_presets.dat

    FAST BootCfg Program Prototype:
    $_bootcfg.dat

    FAST Segger:
    $_segger.csv
    """

    def __init__(self, sys_id, tilegrid, fnn_to_bid, bid_to_ball):
        self.sys_id = str(sys_id)
        self.tilegrid = tilegrid
        self.fnn_to_bid = fnn_to_bid
        self.bid_to_ball = bid_to_ball


[docs]    def write_output(self):
        """Wrapper method to run all writer methods
        Creates directories if they do not exist
        Params:
        Returns:
        """

        if not os.path.exists(WriterConstant.PATH + self.sys_id):
            os.makedirs(WriterConstant.PATH + self.sys_id)
        self.write_to_row()
        self.write_to_col()
        self.write_to_ubump()
        self.write_to_presets_vs()
        self.write_to_presets_dat()
        self.write_to_bootcfg_dat()
        self.write_to_segger_csv()
        return

[docs]    def write_to_row(self):
        """Writes all rowbit associations into GPH file
        Args:
        Returns:
            file: <sys_id>_row.gph
        """

        fname = self.get_filename("row.gph")
        fopen = open(fname, "w")
        for row in xrange(Constant.ROW_SIZE):
            for bit in xrange(Constant.ROWBIT_SIZE):
                rowbit = self.tilegrid.rows[row].bits[bit]
                if rowbit.is_occupied:
                    fnn = rowbit.fnn
                    snn = self.fnn_to_snn(fnn)
                    fopen.write("%s,%d,%d,%d\n" % (snn, fnn, row, bit))
        fopen.close()
        return

[docs]    def fnn_to_snn(self, fnn):
        """Get the System Name associated to FNN
        Args:
            fnn (int)
        Returns:
            str: System Net Name
        """

        if fnn == Constant.FNN_GND:
            return "VSS"
        elif fnn == Constant.FNN_VDD:
            return "VDD"

        bid_list = self.fnn_to_bid[fnn]
        ball = self.bid_to_ball[bid_list[0]]

        return ball.snn



[docs]    def write_to_presets_vs(self):
        """Writes tile information of all non-default tiles into VS file
        Args:
        Returns:
            file: <sys_id>_presets.vs
        """

        fname = self.get_filename("presets.vs")
        ftemp = open("pnr/router/utility/presets_template.txt", "r")
        fopen = open(fname, "w")

        output = Template(ftemp.read())
        data = []

        for row in xrange(Constant.ROW_SIZE):
            for col in xrange(Constant.COL_SIZE):
                tile = self.tilegrid.tiles[row][col]
                if not tile.is_default():
                    data.append("\tpad_preset[%d][%d] = 10'b%s;" % (row, col, tile.get_code()))
                    data.append("\t// %s\n" % tile.gridcode_to_string())
        temp = {'data': "".join(data)}
        output = output.substitute(temp)
        fopen.write(output)
        fopen.close()
        return

[docs]    def write_to_bootcfg_dat(self):
        """Writes bootcfg data of all non-default tiles into DAT file
        Args:
        Returns:
            file: <sys_id>_bootcfg.dat
        """

        fname = self.get_filename("bootcfg.dat")
        fopen = open(fname, "w")
        instr_addr = 0
        instr_addr, instr_data = bootcfg_write_signature(instr_addr)
        fopen.write(instr_data)

        for row in xrange(Constant.ROW_SIZE):
            for col in xrange(Constant.COL_SIZE):
                tile = self.tilegrid.tiles[row][col]
                if not tile.is_default():
                    csel = tile.get_colcode()
                    rsel = tile.get_rowcode()
                    pusel = tile.get_pupcode()
                    pdsel = tile.get_pdncode()

                    instr_addr, instr_data = \
                        bootcfg_write_tilepgm(instr_addr, row, col, pdsel, pusel, rsel, csel)
                    fopen.write(instr_data)

        instr_addr, instr_data = bootcfg_write_done(instr_addr)
        fopen.write(instr_data)
        fopen.close()
        return



[docs]    def write_to_segger_csv(self):
        """Writes segger programming info of all tiles into CSV file
        Args:
        Returns:
            file: <sys_id>_segger.csv
        """

        fname = self.get_filename("segger.csv")
        fopen = open(fname, "w")
        for row in xrange(Constant.ROW_SIZE):
            for col in xrange(Constant.COL_SIZE):
                tile = self.tilegrid.tiles[row][col]
                addr = row << 8 | col << 1
                fopen.write("%d,%s\n" % (addr, tile.get_code()))
        fopen.close()
        return




[docs]def bootcfg_write_signature(addr):
    """Formulate first instruction frame for bootcfg file
    Args:
        addr (int)
    Returns:
        str: Instructions in binary
    """

    # First Instruction Frame, the Signature
    # 6'b011011
    result = [bootcfg_str_builder(addr, WriterConstant.CODE_ONES_BYTE +
                                  WriterConstant.CODE_ONES_BYTE)]
    addr += 2
    result.append(bootcfg_str_builder(addr, WriterConstant.CODE_ONES_BYTE +
                                      WriterConstant.CODE_ZERO_BYTE))
    addr += 2
    result.append(bootcfg_str_builder(addr, WriterConstant.CODE_ZERO_BYTE +
                                      WriterConstant.CODE_ONES_BYTE))
    addr += 2
    return addr, "".join(result)




[docs]def bootcfg_write_tilepgm(addr, row, col, pdsel, pusel, rsel, csel):
    """Formulate tile instruction frame for bootcfg file
    Args:
        addr (int)
        row (int)
        col (int)
        pdsel (int)
        pusel (int)
        rsel (int)
        csel (int)
    Returns:
        str: Instructions in binary
    """

    tile_addr = row << 8 | col << 1
    # First Word - Opc, Hdr
    result = [bootcfg_str_builder(addr, WriterConstant.CODE_BOOT_TILEPGM +
                                  WriterConstant.CODE_ONES_BYTE)]
    addr += 2
    # Second Word - 1b PAD, tile_addr
    result.append(bootcfg_str_builder(addr, "0" + bootcfg_tile_addr_to_str(tile_addr)))
    addr += 2
    # Third Word - 6b PAD, PDSEL, PUSEL, RSEL, CSEL
    result.append(bootcfg_str_builder(addr, "000000"+str(pdsel)+str(pusel)+str(rsel)+str(csel)))
    addr += 2
    return addr, "".join(result)




[docs]def bootcfg_write_done(addr):
    """Formulate last instruction frame for bootcfg file
    Args:
        addr (int)
    Returns:
        str: Instructions in binary
    """

    # First Word - Opc + Hdr
    result = [bootcfg_str_builder(addr, WriterConstant.CODE_BOOT_DONE +
                                  WriterConstant.CODE_ONES_BYTE)]
    addr += 2
    # Second Word - bypass
    result.append(bootcfg_str_builder(addr, WriterConstant.CODE_ZERO_BYTE +
                                      WriterConstant.CODE_ZERO_BYTE))
    addr += 2
    # Third Word - 16b PAD
    result.append(bootcfg_str_builder(addr, WriterConstant.CODE_ZERO_BYTE +
                                      WriterConstant.CODE_ZERO_BYTE))
    addr += 2
    return addr, "".join(result)
