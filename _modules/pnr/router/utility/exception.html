[docs]class InputError(Error):
    """
    InputError: Occurs when there is something wrong with the input
    Possible Scenarios:
        Packer is not run, ubumpid not assigned to netlist
    """
    def __init__(self, msg):
        super(Error)
        self.msg = msg

    def __str__(self):
        return repr(self.msg)
