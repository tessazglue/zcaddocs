"""
Database Schema for System Application,
which consists of Sys, SysComponent and SysNetlist classes.
"""

from __future__ import unicode_literals

from django.db import models
from chiplet.models import Chiplet, Config, ConfigIO


class SysQuerySet(models.QuerySet):

    def not_run(self):
        return self.filter(run_state=0)

    def in_progress(self):
        return self.filter(run_state=1)

    def ran(self):
        return self.filter(run_state=2)



[docs]class Sys(models.Model):
    """
    Stores a system and its checks.

    Custom model methods:
        get_syscomponents(): returns a QuerySet of SysComponents of that system
    """

    SIZE_ENUM = (
        ('S', '1x1 Form Factor'),
        ('M', '1x2 Form Factor'),
        ('L', '2x2 Form Factor'),
    )

    sys_id = models.AutoField(primary_key=True)
    sys_name = models.CharField("Name of sys configuration", max_length=30)
    size = models.CharField(max_length=1, choices=SIZE_ENUM)
    run_state = models.IntegerField(default=0)
    packer_check = models.IntegerField(default=0)
    router_check = models.IntegerField(default=0)
    bucket = models.CharField(max_length=300, blank=True)
    objects = SysQuerySet.as_manager()

    def __unicode__(self):
        return str(self.sys_id) + ", " + self.sys_name

    def __str__(self):
        return unicode(self).encode('utf-8')

    def get_syscomponents(self):
        return SysComponent.objects.filter(sys=self)




[docs]class SysComponent(models.Model):
    """
    Stores a system component and its packing information.
    Foreign keys to a chiplet, a config and a sys.

    Custom model methods:
        get_sysnetlist(): returns a QuerySet of SysNetlist of that SysComponent
    """

    syscomp_id = models.AutoField(primary_key=True)
    sys = models.ForeignKey(Sys, related_name='syscomp', on_delete=models.CASCADE)
    chiplet = models.ForeignKey(Chiplet, on_delete=models.CASCADE)
    cfg = models.ForeignKey(Config, on_delete=models.CASCADE)
    cen_x = models.IntegerField("x-coord of center of chiplet on FAST")
    cen_y = models.IntegerField("y-coord of center of chiplet on FAST")
    rot = models.IntegerField("rotation of chiplet on FAST in degrees")
    bucket = models.CharField(max_length=300, blank=True)

    def __unicode__(self):
        return str(self.chiplet.mpn)

    def __str__(self):
        return unicode(self).encode('utf-8')

    def get_sysnetlist(self):
        return SysNetlist.objects.filter(syscomp=self)




[docs]class SysNetlist(models.Model):
    """
    Stores a system IO and its information.
    Foreign keys to a syscomponent and a configIO.
    """

    sysnet_id = models.AutoField(primary_key=True)
    syscomp = models.ForeignKey(SysComponent, on_delete=models.CASCADE)
    cfgio = models.ForeignKey(ConfigIO, on_delete=models.CASCADE)
    ubumpid = models.IntegerField(blank=True)
    bucket = models.CharField(max_length=300, blank=True)

    def __unicode__(self):
        return str(self.syscomp)

    def __str__(self):
        return unicode(self).encode('utf-8')
